The Calculation Engine (src/utils/pricingCalc.js)
jsimport { PRICING_MATRIX } from '../data/pricingMatrix';

/**
 * Pure function — no side effects, no React imports, fully testable.
 * @param {string} tier     — 'starter' | 'pro' | 'enterprise'
 * @param {string} currency — 'USD' | 'INR' | 'EUR'
 * @param {string} cycle    — 'monthly' | 'annual'
 * @returns {{ symbol: string, amount: string, period: string, badge: string|null }}
 */
export function computePrice(tier, currency, cycle) {
  const { baseUSDCents }          = PRICING_MATRIX.tiers[tier];
  const { multiplier, badge }     = PRICING_MATRIX.cycles[cycle];
  const { symbol, tariff, locale } = PRICING_MATRIX.currencies[currency];

  const raw     = (baseUSDCents / 100) * multiplier * tariff;
  const amount  = new Intl.NumberFormat(locale, {
    minimumFractionDigits: 0,
    maximumFractionDigits: 0,
  }).format(raw);

  return {
    symbol,
    amount,
    period: cycle === 'annual' ? '/mo, billed annually' : '/month',
    badge,
  };
}

Architecture: Pricing Island Pattern
jsx// src/components/Pricing.jsx
// This component is the isolation boundary.
// It uses useRef for currency/cycle to avoid re-rendering PricingCard siblings.

import { useRef, useCallback } from 'react';
import { PRICING_MATRIX } from '../data/pricingMatrix';
import { computePrice } from '../utils/pricingCalc';
import PricingCard from './ui/PricingCard';

export default function Pricing() {
  // useRef — NOT useState — so toggling never triggers a re-render of this component
  const currencyRef = useRef('USD');
  const cycleRef    = useRef('monthly');

  // Direct DOM writes — bypasses React reconciler entirely
  const updateAllPrices = useCallback(() => {
    const currency = currencyRef.current;
    const cycle    = cycleRef.current;

    Object.keys(PRICING_MATRIX.tiers).forEach((tier) => {
      const { symbol, amount, period, badge } = computePrice(tier, currency, cycle);

      // Target the exact text nodes — nothing else in the DOM changes
      const priceEl  = document.getElementById(`price-amount-${tier}`);
      const symbolEl = document.getElementById(`price-symbol-${tier}`);
      const periodEl = document.getElementById(`price-period-${tier}`);
      const badgeEl  = document.getElementById(`price-badge-${tier}`);

      if (priceEl)  priceEl.textContent  = amount;
      if (symbolEl) symbolEl.textContent = symbol;
      if (periodEl) periodEl.textContent = period;
      if (badgeEl) {
        badgeEl.textContent = badge ?? '';
        badgeEl.style.opacity = badge ? '1' : '0';
      }
    });
  }, []);

  const handleCycleChange = useCallback((newCycle) => {
    cycleRef.current = newCycle;
    // Update toggle UI manually
    document.getElementById('toggle-monthly')
      ?.setAttribute('data-active', newCycle === 'monthly');
    document.getElementById('toggle-annual')
      ?.setAttribute('data-active', newCycle === 'annual');
    updateAllPrices();
  }, [updateAllPrices]);

  const handleCurrencyChange = useCallback((e) => {
    currencyRef.current = e.target.value;
    updateAllPrices();
  }, [updateAllPrices]);

  return (
    <section id="pricing" aria-labelledby="pricing-heading">
      {/* Controls */}
      <div className="pricing-controls">
        {/* Cycle toggle */}
        <div role="group" aria-label="Billing cycle">
          <button id="toggle-monthly" data-active="true"
            onClick={() => handleCycleChange('monthly')}>Monthly</button>
          <button id="toggle-annual"  data-active="false"
            onClick={() => handleCycleChange('annual')}>Annual</button>
        </div>

        {/* Currency selector */}
        <select onChange={handleCurrencyChange} aria-label="Select currency"
          defaultValue="USD">
          <option value="USD">$ USD</option>
          <option value="INR">₹ INR</option>
          <option value="EUR">€ EUR</option>
        </select>
      </div>

      {/* Cards — rendered once, never re-rendered */}
      <div className="pricing-grid">
        {Object.entries(PRICING_MATRIX.tiers).map(([key, tier]) => {
          const initial = computePrice(key, 'USD', 'monthly');
          return (
            <PricingCard
              key={key}
              tierId={key}
              tier={tier}
              initialPrice={initial}
            />
          );
        })}
      </div>
    </section>
  );
}

FEATURE 2 — BENTO-TO-ACCORDION WITH CONTEXT LOCK (10 pts)
The Hook (src/hooks/useBreakpoint.js)
jsimport { useState, useEffect } from 'react';

const MOBILE_BREAKPOINT = 768;

/**
 * Uses ResizeObserver on document.body for precision.
 * Returns boolean — true when viewport < 768px.
 */
export function useBreakpoint() {
  const [isMobile, setIsMobile] = useState(
    () => window.innerWidth < MOBILE_BREAKPOINT
  );

  useEffect(() => {
    const observer = new ResizeObserver((entries) => {
      for (const entry of entries) {
        setIsMobile(entry.contentRect.width < MOBILE_BREAKPOINT);
      }
    });
    observer.observe(document.body);
    return () => observer.disconnect();
  }, []);

  return isMobile;
}

jsximport { useRef, useState, useEffect, useCallback } from 'react';
import { useBreakpoint } from '../hooks/useBreakpoint';
import { FEATURES } from '../data/featuresData';

export default function Features() {
  // Tracks which bento card is currently hovered/active on desktop
  const activeIndexRef  = useRef(null);

  // Accordion open index — null means all closed
  const [accordionOpen, setAccordionOpen] = useState(null);

  const isMobile = useBreakpoint();

  // THE CONTEXT LOCK — transfers desktop hover state to mobile accordion
  // Only fires when crossing the breakpoint threshold
  const prevIsMobileRef = useRef(isMobile);
  useEffect(() => {
    const crossedToMobile = !prevIsMobileRef.current && isMobile;
    if (crossedToMobile && activeIndexRef.current !== null) {
      setAccordionOpen(activeIndexRef.current);
    }
    prevIsMobileRef.current = isMobile;
  }, [isMobile]);

  const handleBentoEnter = useCallback((id) => {
    activeIndexRef.current = id;
  }, []);

  const handleBentoLeave = useCallback(() => {
    // Do NOT clear activeIndexRef — it must persist for the context lock
  }, []);

  const toggleAccordion = useCallback((id) => {
    setAccordionOpen((prev) => (prev === id ? null : id));
  }, []);

  if (isMobile) {
    return (
      <section id="features" aria-labelledby="features-heading">
        <h2 id="features-heading">Everything your data pipeline needs</h2>
        <div className="accordion">
          {FEATURES.map((f) => {
            const isOpen = accordionOpen === f.id;
            return (
              <article key={f.id} className="accordion-item">
                <button
                  className="accordion-trigger"
                  onClick={() => toggleAccordion(f.id)}
                  aria-expanded={isOpen}
                  aria-controls={`accordion-panel-${f.id}`}
                  id={`accordion-btn-${f.id}`}
                >
                  <span className="accordion-icon">{/* SVG here */}</span>
                  <span>{f.title}</span>
                  <span className={`accordion-chevron ${isOpen ? 'open' : ''}`}>
                    {/* chevron-down.svg */}
                  </span>
                </button>
                <div
                  id={`accordion-panel-${f.id}`}
                  role="region"
                  aria-labelledby={`accordion-btn-${f.id}`}
                  className={`accordion-panel ${isOpen ? 'accordion-panel--open' : ''}`}
                >
                  <p>{f.body}</p>
                </div>
              </article>
            );
          })}
        </div>
      </section>
    );
  }

  return (
    <section id="features" aria-labelledby="features-heading">
      <h2 id="features-heading">Everything your data pipeline needs</h2>
      <div className="bento-grid" role="list">
        {FEATURES.map((f) => (
          <article
            key={f.id}
            className={`bento-card bento-card--${f.size ?? 'md'}`}
            role="listitem"
            onMouseEnter={() => handleBentoEnter(f.id)}
            onMouseLeave={handleBentoLeave}
            onFocus={() => handleBentoEnter(f.id)}
            tabIndex={0}
          >
            <span className="bento-icon">{/* SVG */}</span>
            <h3>{f.title}</h3>
            <p>{f.body}</p>
          </article>
        ))}

  );
